# Day 2 Commands

# Create a folder and go into it
mkdir day2_folder
cd day2_folder

# Create files
touch a.txt b.txt c.log

# Copy a file
cp a.txt backup.txt

# Rename/move a file
mv b.txt renamed.txt

# Delete a file
rm c.log

# Delete a folder recursively
cd ..
rm -r day2_folder

# List files with details
ls -l
